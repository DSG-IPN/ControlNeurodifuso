function [R,x1]=aprend_regio(a1,a2,mu)
A=0;
epocas=0;
w=[1;1;0];
[f,c]=size(a1);
while A<(c*2)
A=0;
for i=1:c
fd=w.'*[a1(1:2,i);mu];%Como se debe usar w.'*[a1(1:3,i);mu];
%comparamos fd
if fd>0
A=A+1;
else
w=w+[a1(1:2,i);mu];
A=0;
end
fd=w.'*[a2(1:2,i);mu];
%comparamos fd
if fd<0
A=A+1;
else
w=w-[a2(1:2,i);mu];
A=0;
end
end
epocas=epocas+1;
end
x1=sym('x1');
x2=sym('x2');
x=[x1 x2 1];
R=x*w;
x1=solve(R,x1);
%fplot(x1,[0 300])
epocas
end
